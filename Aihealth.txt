    <script>
        // Set initial dark mode based on user preference or local storage
        document.addEventListener('DOMContentLoaded', () => {
            const isDarkMode = localStorage.getItem('darkMode') === 'true';
            if (isDarkMode) {
                document.body.classList.add('dark-mode');
                document.getElementById('darkIcon').textContent = '☀';
            }
        });

        function toggleDarkMode() {
            const body = document.body;
            const icon = document.getElementById('darkIcon');
            const isCurrentlyDarkMode = body.classList.toggle("dark-mode");
            icon.textContent = isCurrentlyDarkMode ? '☀' : '🌙';
            localStorage.setItem('darkMode', isCurrentlyDarkMode); // Save preference
        }

        function toggleProfileDropdown() {
            const dropdown = document.getElementById('profileDropdown');
            dropdown.classList.toggle('active');
        }

        // Close dropdown if clicked outside
        window.onclick = function(event) {
            if (!event.target.matches('.profile-icon') && !event.target.matches('.profile span')) {
                const dropdowns = document.getElementsByClassName("profile-dropdown");
                for (let i = 0; i < dropdowns.length; i++) {
                    const openDropdown = dropdowns[i];
                    if (openDropdown.classList.contains('active')) {
                        openDropdown.classList.remove('active');
                    }
                }
            }
        }

        async function analyzeHealth() {
            const fileInput = document.getElementById('health-report');
            const symptoms = document.getElementById('symptoms').value;
            const resultDiv = document.getElementById('result');
            const loadingSpinner = document.getElementById('loadingSpinner');

            // Clear previous results and show loading spinner
            resultDiv.innerHTML = '';
            loadingSpinner.style.display = 'block';

            if (!fileInput.files.length || !symptoms.trim()) { // Added trim() for symptoms
                resultDiv.innerHTML = '<span style="color: red;">Please upload a health report and describe your symptoms.</span>';
                loadingSpinner.style.display = 'none'; // Hide spinner on validation error
                return;
            }

            const formData = new FormData();
            formData.append('health-report', fileInput.files[0]);
            formData.append('symptoms', symptoms);

            try {
                const response = await fetch('/api/analyze-health', {
                    method: 'POST',
                    body: formData
                });

                const data = await response.json();

                if (!response.ok) {
                    throw new Error(data.error || 'Failed to analyze health data.');
                }

                // Display results using template literals
                resultDiv.innerHTML = `
                    <p><strong>Analysis Results:</strong></p>
                    <p><strong>File Uploaded:</strong> ${data.filename}</p>
                    <p><strong>Symptoms Analyzed:</strong> ${data.symptoms}</p>
                    <p><strong>Possible Condition:</strong> ${data.condition}</p>
                    <p><strong>Medicine Suggestions:</strong> ${data.suggestions}</p>
                    <p style="font-style: italic; color: #e74c3c;"><strong>Disclaimer:</strong> ${data.disclaimer}</p>
                `;
            } catch (error) {
                // Corrected string interpolation for error message
                resultDiv.innerHTML = `<span style="color: red;">Error: ${error.message}</span>`;
            } finally {
                loadingSpinner.style.display = 'none'; // Hide spinner after request completes
            }
        }
    </script>
